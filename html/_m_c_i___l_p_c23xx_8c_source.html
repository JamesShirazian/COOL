00673

00675       rstat = mci_command (SEND_CSD, arg, RESP_LONG, &rval[0]);
00676       if (rstat == 0) {
00677          /* Response is back and correct. */
00678          break;
00679       }
00680    }
00681    if (i == 0) {
00682       /* Read CSD failed. */
00683       return (__FALSE);
00684    }
00685 
00686    /* Read Block length. */
00687    v = (rval[1] >> 16) & 0x0F;
00688    cfg->read_blen = 1 << v;
00689 
00690    /* Write Block length */
00691    v = (rval[3] >> 22) & 0x0F;
00692    cfg->write_blen = 1 << v;
00693
