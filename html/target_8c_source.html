00078 ** Function name:               ConfigurePLL
00079 **
00080 ** Descriptions:                Configure PLL switching to main OSC instead of IRC
00081 **                                              at power up and wake up from power down. 
00082 **                                              This routine is used in TargetResetInit() and those
00083 **                                              examples using power down and wake up such as
00084 **                                              USB suspend to resume, ethernet WOL, and power management
00085 **                                              example
00086 ** parameters:                  None
00087 ** Returned value:              None
00088 ** 
00089 ******************************************************************************/
00090 void ConfigurePLL ( void )
00091 {
00092   DWORD MValue, NValue;
00093 
00094   if ( PLLSTAT & (1 << 25) )
00095   {
00096         PLLCON = 1;                     /* Enable PLL, disconnected */
00097         PLLFEED = 0xaa;
00098         PLLFEED = 0x55;
00099   }
00100 
00101   PLLCON = 0;                           /* Disable PLL, disconnected */
00102   PLLFEED = 0xaa;
00103   PLLFEED = 0x55;
00104     
00105   SCS |= 0x20;                  /* Enable main OSC */
00106   while( !(SCS & 0x40) );       /* Wait until main OSC is usable */
00107 
00108   CLKSRCSEL = 0x1;              /* select main OSC, 12MHz, as the PLL clock source */
00109 
00110   PLLCFG = PLL_MValue | (PLL_NValue << 16);
00111   PLLFEED = 0xaa;
00112   PLLFEED = 0x55;
00113       
00114   PLLCON = 1;                           /* Enable PLL, disconnected */
00115   PLLFEED = 0xaa;
00116   PLLFEED = 0x55;
00117 
00118   CCLKCFG = CCLKDivValue;       /* Set clock divider */
00119 #if USE_USB
00120   USBCLKCFG = USBCLKDivValue;           /* usbclk = 288 MHz/6 = 48 MHz */
00121 #endif
00122 
00123   while ( ((PLLSTAT & (1 << 26)) == 0) );       /* Check lock bit status */
00124     
00125   MValue = PLLSTAT & 0x00007FFF;
00126   NValue = (PLLSTAT & 0x00FF0000) >> 16;
00127   while ((MValue != PLL_MValue) && ( NValue != PLL_NValue) );
00128 
00129   PLLCON = 3;                           /* enable and connect */
00130   PLLFEED = 0xaa;
00131   PLLFEED = 0x55;
00132   while ( ((PLLSTAT & (1 << 25)) == 0) );       /* Check connect bit status */
00133   return;
00134 }
